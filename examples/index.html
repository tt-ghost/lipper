Light weight ripple library

Usage:

Selector support querySelector. detailed documentation.

lipper.init({
  // Duration. The unit is second. Default value: 1.5
  duration: 1.5,
  // CSS selector. String or string array. No default value. Required
  selector: ['button', '.container'],
  // Diffusion radius. The unit is px. Default value: 50
  radius: 50,
  // z-index value. Default value: 1000
  zindex: 1000,
  // Whether lipper in the container. Default value: false
  center: true,
  // Whether to continue lipper outside the container.
  overflow: false,
  // The color of lipper. Default value: rgba(255, 0, 0, .5)
  color: 'rgba(255, 0, 0, .5)'
})
